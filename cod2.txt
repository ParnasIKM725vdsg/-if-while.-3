#include <iostream>
using namespace std;
int main()
{
   int n = 10;
   while (n <= 99) {
       int d = n / 10;
       int rem = n % 10;
       int suma = d + rem;
       int differ = d - rem;
       if (suma + differ * differ == n) {
           cout << n << endl;
       }
       n++;
   }
   return 0;
}
