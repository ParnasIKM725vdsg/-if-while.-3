#include <iostream>
using namespace std;
int main()
{
    double a, b, c;
    cout << "Enter three numbers: ";
    cin >> a >> b >> c;

    double d = a;
    if (b > d) d = b;
    if (c > d) d = c;

    double s = a;
    if (b < s) s = b;
    if (c < s) s = c;

    cout << "Largest (a) = " << d << endl;
    cout << "Smallest (x) = " << s << endl;
    return 0;
}
